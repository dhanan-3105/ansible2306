  hosts: all
  remote_user: opc
  vars:
    x: " "
  tasks:
   - name: Read the file
     command: cat /home/opc/.ssh/authorized_keys
     register: x

   - name: print file
     debug:
      msg: "the content of file is {{ x }}"

   - name: write a file
     file: path=/home/opc/configfiles state=touch


   - name: closure
     debug:
        msg : "file created in {{inventory_hostname}}"
